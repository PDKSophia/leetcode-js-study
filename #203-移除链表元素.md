## 移除链表元素
删除链表中等于给定值 val 的所有节点。

```base
示例:

输入: 1->2->6->3->4->5->6, val = 6
输出: 1->2->3->4->5
```

```javascript
  /**
   * Definition for singly-linked list.
   * function ListNode(val) {
   *     this.val = val;
   *     this.next = null;
   * }
  */

  /**
   * @param {ListNode} head
   * @param {number} val
   * @return {ListNode}
  */
  function removeElement = function (head, val) {
    var arr = []
    while (head) {
      if (head.val !== val) {
        arr.push(head.val)
      }
      head = head.next
    }
    return arr
  }
```
